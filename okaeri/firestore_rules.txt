rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // ---------- Helpers ----------
    function isSignedIn() {
      return request.auth != null;
    }
    function isOwner(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }
    function isCoupleMember(coupleId) {
      return isSignedIn() &&
        request.auth.uid in get(/databases/$(database)/documents/couples/$(coupleId)).data.members;
    }
    // A note is visible to me if it's shared, or if it's my own private note.
    function canSeeNote(noteData) {
      return noteData.visibility == 'shared' ||
        noteData.visibility == 'private:' + request.auth.uid;
    }
    // True if userId and the caller belong to the same couple.
    // Relies on each user doc storing its own coupleId (set when an
    // invite is accepted).
    function isPartnerOf(userId) {
      return isSignedIn() &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        exists(/databases/$(database)/documents/users/$(userId)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.coupleId != null &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.coupleId ==
        get(/databases/$(database)/documents/users/$(userId)).data.coupleId;
    }
    // Either you, or your partner.
    function canAccessUser(userId) {
      return isOwner(userId) || isPartnerOf(userId);
    }

    // ---------- users/{userId} ----------
    match /users/{userId} {
      // Partner can read/edit your profile, but can't be the one to
      // set/clear coupleId on you (that's reserved for you or your
      // invite-acceptance flow) — prevents a partner from silently
      // unlinking you.
      allow read: if canAccessUser(userId);
      allow update: if canAccessUser(userId) &&
        (isOwner(userId) ||
          !('coupleId' in request.resource.data.diff(resource.data).affectedKeys()));
      allow create: if isOwner(userId);

      match /profileNotes/{noteId} {
        // canSeeNote already keys private notes to request.auth.uid,
        // so a partner naturally can't see/write your private ones —
        // no extra logic needed here.
        allow read: if canAccessUser(userId) && canSeeNote(resource.data);
        allow create: if canAccessUser(userId) && canSeeNote(request.resource.data);
        allow update: if canAccessUser(userId) &&
          canSeeNote(resource.data) && canSeeNote(request.resource.data);
        allow delete: if canAccessUser(userId) && canSeeNote(resource.data);
      }
    }

    // ---------- couples/{coupleId} ----------
    match /couples/{coupleId} {
      allow read: if isCoupleMember(coupleId);
      allow create: if isSignedIn() &&
        request.resource.data.members.size() == 1 &&
        request.resource.data.members[0] == request.auth.uid;
      // Split into two allowed shapes:
      //  1. Joining: 1 member -> 2 members, caller added, no one removed.
      //  2. Ordinary edits by an existing member: membership itself can't change.
      allow update: if
        (
          isSignedIn() &&
          resource.data.members.size() == 1 &&
          request.resource.data.members.size() == 2 &&
          request.auth.uid in request.resource.data.members &&
          !(request.auth.uid in resource.data.members)
        ) ||
        (
          isCoupleMember(coupleId) &&
          request.resource.data.members == resource.data.members
        );
      allow delete: if isCoupleMember(coupleId);

      // ---------- shared couple collections (fully editable together) ----------
      match /{collection}/{docId} {
        allow read, write: if isCoupleMember(coupleId) &&
          collection in ['messageBoard', 'calendarNotes', 'plans', 'periodTracker', 'meta'];
      }

      // ---------- notes ("Our Room" + "My Corner", split by visibility) ----------
      match /notes/{noteId} {
        allow read: if isCoupleMember(coupleId) && canSeeNote(resource.data);
        allow create: if isCoupleMember(coupleId) &&
          canSeeNote(request.resource.data);
        allow update: if isCoupleMember(coupleId) &&
          canSeeNote(resource.data) &&
          canSeeNote(request.resource.data);
        allow delete: if isCoupleMember(coupleId) &&
          canSeeNote(resource.data);
      }
    }

    // ---------- invites/{code} ----------
    match /invites/{code} {
      allow read: if isSignedIn();
      allow create: if isSignedIn() &&
        request.resource.data.creatorUid == request.auth.uid;
      // Only allow flipping used: false -> true (redeeming), nothing else.
      allow update: if isSignedIn() &&
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['used']) &&
        resource.data.used == false &&
        request.resource.data.used == true;
      allow delete: if false;
    }

    // ---------- Deny everything else by default ----------
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
